(1) 文件名以.c结尾

(2) /* 或 //注释

(3) #include <stdio.h>  引入终端读写数据的代码

(4) puts() 输出内容，只能输出字符串，不能输出整数
    printf() 格式化输出，如 printf("%d,%d",10,20);
	%d   表示整数
	%s   表示字符串
	%f   表示浮点数，默认6位小数，
	%c   表示字符
	%-9d  格式输出，表示至少占9个字符，不足用-对齐
	%.10d 格式输出, 表示占用10个字符，不足在前面加0
	
	scanf()  获取用户输入  如：scanf("%d %d",&a,&b);
	
	字符串结束标志 \0
	
	gets() 获取用户键盘输入的字符串，以上几个函数需要引入stdio.h

(5) 转义字符

(6) 字节和比特  8bit = 1byte  1024 byte = 1kb

(7) 数据类型：整形(2或4字节 分有符号和无符号;0 到 65,535 或 0 到 4,294,967,295)  int a = 10；
              小数(float长度为4个字节，double长度8个字节)
			  字符(始终为1个字节)

(8) sizeof() 计算数据占用的内存空间，以字节计算  如:sizeof(23);

(9) 数组：一维数组：定义数组 int a[4]  每个元素都必须是整形；
					赋值：a[0] = 1；
					数组初始化： int a[4] = {1,2,3,4}或int a[]= {1,3}  不能写成int a[10] = 1;
		  
		  二维数组：定义数组 int a[3][4];  初始化int a[2][4] = {{1,2,3,4},{4,5,6,7}}
		  

(10) 字符串函数：
	获取字符串长度：strlen(), 如strlen(str);
	字符串连接函数:strcat() ,如strcat(str1,str2)
	字符串复制函数:strcpy() ,如 strcpy(str1,str2)
	
(11) 预处理命令，#include   引入文件或库
				 宏定义#define PI 3.14
				 取消宏定义 #undef PI
				 宏定义的变量名，若用引号括起来，则不作替换 如：#define OK 100 printf("OK"); 则直接输出OK
				 宏定义允许嵌套，如 #define PI 3.14 #define S PI*y
				 宏变量名大小写都可以，习惯是大写
				 
				 带参数的宏定义,如 #define M(y) y*y

				条件编译:#define WIN true #ifdef WIN ... #else ... #endif
										  #ifndef WIN ... #else ... #endif
										  #if WIN ... #else ... #endif
										  
(12) 指针

		定义指针变量：  类型说明符 ×变量名；如int *p1, p1是指针变量,表示变量地址
		&取地址符：获取变量名的地址  如：&a； int a = 10, int *p=&a;
		不允许把数字直接赋值给指针变量
