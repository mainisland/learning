・・・
#include <stdio.h>

/**
* 简单的桶排序算法； 
* 描述：有10个同学总分是10分，分数分别是 2,5,3,1,8,9,5,6,7,3 ；从小到大排序 
* 使用桶排序：创建大小为11的数组；a[0~10] ；如果得分多少就在相应的数组上加上1 
*/
int main() {
	int a[11],i,j,t;
	for(i=0;i<=10;i++) {
		a[i] = 0;
	}
	for(i=1;i<=10;i++) {
		scanf("%d",&t);
		a[t]++;
	} 
	for(i=0;i<=10;i++) {
		for(j=1;j<=a[i];j++) {
			printf("%d",i);
		}
	}
	return 0;
}
